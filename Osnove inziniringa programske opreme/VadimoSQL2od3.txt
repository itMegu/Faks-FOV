1. SELECT MAX(SKUPAJ) , MIN(SKUPAJ) , SUM(SKUPAJ) , AVG(SKUPAJ)
   FROM V_NAROCILA;

2. SELECT ST_ODDELKA, MIN(PLACA), MAX(PLACA), SUM(PLACA), AVG(PLACA)
   FROM V_DELAVCI GROUP BY ST_ODDELKA;

3. SELECT NADREJENI.PRIIMEK, COUNT(PODREJENI.EMSO)
   FROM V_DELAVCI PODREJENI
   JOIN V_DELAVCI NADREJENI ON PODREJENI.NADREJENI = NADREJENI.EMSO
   GROUP BY NADREJENI.PRIIMEK;


4. SELECT COUNT(*) 
   FROM V_NAROCILA WHERE DODATEK = 'A';

5. SELECT COUNT(*)
   FROM V_VRSTICE_NAROCIL;

6. SELECT MAX(SKUPAJ) - MIN(SKUPAJ)
   FROM V_NAROCILA;

7. SELECT NAZIV, AVG(STANDARDNA_CENA)
   FROM V_IZDELKI I 
   JOIN V_CENE C ON I.IZDELEK_ID = C.IZDELEK_ID GROUP BY I.NAZIV 
   HAVING AVG(C.STANDARDNA_CENA) < 10;

8. SELECT SUM(SKUPAJ)
   FROM V_NAROCILA N JOIN V_KUPCI K ON N.KUPEC_ID = K.KUPEC_ID 
   WHERE K.NAZIV = 'TRUST TENIS';
 
9. SELECT SUM(KOLICINA) 
   FROM V_VRSTICE_NAROCIL VN 
   JOIN V_IZDELKI VI ON VN.IZDELEK_ID = VI.IZDELEK_ID 
   WHERE VI.NAZIV = 'KNJIGA: "PRVI UDARCI"';

10. SELECT AVG(SKUPAJ) 
    FROM V_NAROCILA 
    WHERE EXTRACT(YEAR FROM DATUM_NAROCILA) = 2011;

11. SELECT AVG(PLACA) 
    FROM V_DELAVCI D 
    JOIN V_NIVOPLACE N ON D.PLACA BETWEEN N.NAJNIZJA_PLACA AND N.NAJVISJA_PLACA 
    WHERE N.NIVO = 1;
