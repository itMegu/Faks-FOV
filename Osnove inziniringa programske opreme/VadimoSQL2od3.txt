1. SELECT MAX(SKUPAJ) AS Maximum, MIN(SKUPAJ) AS Minimum, SUM(SKUPAJ) AS Vsota, AVG(SKUPAJ) AS Povprecje FROM V_NAROCILA;

2. SELECT ST_ODDELKA, MIN(PLACA) AS Minimalna, MAX(PLACA) AS Maksimalna, SUM(PLACA) AS Vsota, AVG(PLACA) AS Povprecja FROM V_DELAVCI GROUP BY ST_ODDELKA;

3. SELECT NADREJENI, COUNT(*) AS St_Delavcev FROM V_DELAVCI WHERE NADREJENI IS NOT NULL GROUP BY NADREJENI;

4. SELECT COUNT(*) AS St_Narocil FROM V_NAROCILA WHERE DODATEK = 'A';

5. SELECT COUNT(*) AS St_Vrstic FROM V_VRSTICE_NAROCIL;

6. SELECT MAX(SKUPAJ) - MIN(SKUPAJ) AS RAZLIKA FROM V_NAROCILA;

7. SELECT I.NAZIV, AVG(C.STANDARDNA_CENA) AS Povprecna_Cena FROM V_IZDELKI I JOIN V_CENE C ON I.IZDELEK_ID = C.IZDELEK_ID GROUP BY I.NAZIV HAVING AVG(C.STANDARDNA_CENA) < 10;

8. SELECT SUM(SKUPAJ) AS Skupno_Zapravljen FROM V_NAROCILA N JOIN V_KUPCI K ON N.KUPEC_ID = K.KUPEC_ID WHERE K.NAZIV = 'TRUST TENIS';
 
9. SELECT SUM(KOLICINA) AS Prodanih_Knjig FROM V_VRSTICE_NAROCIL VN JOIN V_IZDELKI VI ON VN.IZDELEK_ID = VI.IZDELEK_ID WHERE VI.NAZIV = 'KNJIGA: "PRVI UDARCI"';

10. SELECT AVG(SKUPAJ) AS Povprecna_Vrednost FROM V_NAROCILA WHERE EXTRACT(YEAR FROM DATUM_NAROCILA) = 2011;

11. SELECT AVG(D.PLACA) AS Povprecna_Placa FROM V_DELAVCI D JOIN V_NIVOPLACE N ON D.PLACA BETWEEN N.NAJNIZJA_PLACA AND N.NAJVISJA_PLACA WHERE N.NIVO = 1;
